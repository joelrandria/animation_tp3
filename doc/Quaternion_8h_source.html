53  TQuaternion(const Vec3Real& from, const Vec3Real& to)
54  {
55  const Real epsilon = 1E-10f;
56 
57  const Real fromSqNorm = from.squaredNorm();
58  const Real toSqNorm = to.squaredNorm();
59  // Identity TQuaternion when one vector is null
60  if ((fromSqNorm < epsilon) || (toSqNorm < epsilon))
61  {
62  q[0]=q[1]=q[2]=0.0;
63  q[3]=1.0;
64  }
65  else
66  {
67  Vec3Real axis = cross(from, to);
68  const Real axisSqNorm = axis.squaredNorm();
69 
70  // Aligned vectors, pick any axis, not aligned with from or to
71  if (axisSqNorm < epsilon)
72  {
73  axis = Vec3Real(1.0, 0.0, 0.0);
74  if (axis*from < 0.1*sqrt(fromSqNorm))
75  axis = Vec3Real(0.0, 1.0, 0.0);
76  }
77 
78  Real angle = asin(sqrt(axisSqNorm / (fromSqNorm * toSqNorm)));
79 
80  if (from*to < 0.0)
81  angle = M_PI-angle;
82 
83  setAxisAngle(axis, angle);
84  }
85  }

135  void setFromRotationMatrix(const Real m[3][3])
136  {
137  // First, find largest diagonal in matrix:
138  int i = 2;
139  if (m[0][0] > m[1][1])
140  {
141  if (m[0][0] > m[2][2])
142  {
143  i = 0;
144  }
145  }
146  else
147  {
148  if (m[1][1] > m[2][2])
149  {
150  i = 1;
151  }
152  }
153 
154  if (m[0][0]+m[1][1]+m[2][2] > m[i][i])
155  {
156  // Compute w first:
157  q[3] = sqrt(m[0][0]+m[1][1]+m[2][2]+1.0)/2.0;
158  // And compute other values:
159  q[0] = (m[2][1]-m[1][2])/(4.0*q[3]);
160  q[1] = (m[0][2]-m[2][0])/(4.0*q[3]);
161  q[2] = (m[1][0]-m[0][1])/(4.0*q[3]);
162  }
163  else
164  {
165  // Compute x, y, or z first:
166  int j = (i+1)%3;
167  int k = (i+2)%3;
168 
169  // Compute first value:
170  q[i] = sqrt(m[i][i]-m[j][j]-m[k][k]+1.0)/2.0;
171 
172  // And the others:
173  q[j] = (m[i][j]+m[j][i])/(4.0*q[i]);
174  q[k] = (m[i][k]+m[k][i])/(4.0*q[i]);
175 
176  q[3] = (m[k][j]-m[j][k])/(4.0*q[i]);
177  }
178  }

486  static TQuaternion slerp(const TQuaternion& a, const TQuaternion& b, Real t, bool allowFlip=true)
487  {
488  Real cosAngle = TQuaternion::dot(a, b);
489 
490  Real c1, c2;
491  // Linear interpolation for close orientations
492  if ((1.0 - fabs(cosAngle)) < 0.01)
493  {
494  c1 = 1.0 - t;
495  c2 = t;
496  }
497  else
498  {
499  // Spherical interpolation
500  Real angle = acos(fabs(cosAngle));
501  Real sinAngle = sin(angle);
502  c1 = sin(angle * (1.0 - t)) / sinAngle;
503  c2 = sin(angle * t) / sinAngle;
504  }
505 
506  // Use the shortest path
507  if (allowFlip && (cosAngle < 0.0))
508  c1 = -c1;
509 
510  return TQuaternion(c1*a[0] + c2*b[0], c1*a[1] + c2*b[1], c1*a[2] + c2*b[2], c1*a[3] + c2*b[3]);
511  }
