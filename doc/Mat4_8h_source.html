746  template< typename T >
747  Mat4< T >& Mat4< T >::inverse()
748  {
749  Mat4< T >& mat( *this );
750  unsigned int i,j,k;
751  // Locations of pivot elements
752  unsigned int *pvt_i, *pvt_j;
753 
754  T pvt_matal; // matalue of current pimatot element
755  T hold; // Temporary storage
756  T determ; // Determinant
757 
758  determ = 1.0;
759 
760  pvt_i = new unsigned int[4];
761  pvt_j = new unsigned int[4];
762 
763  for (k = 0; k < 4; k++)
764  {
765  // Locate k'th pivot element
766  pvt_matal = mat.vv[k][k]; // Initialize for search
767  pvt_i[k] = k;
768  pvt_j[k] = k;
769  for (i = k; i < 4; i++)
770  for (j = k; j < 4; j++)
771  if (fabs(mat.vv[i][j]) > fabs(pvt_matal))
772  {
773  pvt_i[k] = i;
774  pvt_j[k] = j;
775  pvt_matal = mat.vv[i][j];
776  }
777  // Product of pivots, gives determinant when finished
778  determ *= pvt_matal;
779  if (determ == 0.0)
780  {
781  // matrix is singular (zero determinant).
782  delete[] pvt_i;
783  delete[] pvt_j;
784  return mat.setIdentity();
785  }
786 
787  // "Interchange" rows (with sign change stuff)
788  i = pvt_i[k];
789  if (i != k) // If rows are different
790  for (j = 0; j < 4; j++)
791  {
792  hold = -mat.vv[k][j];
793  mat.vv[k][j] = mat.vv[i][j];
794  mat.vv[i][j] = hold;
795  }
796 
797  // "Interchange" columns
798  j = pvt_j[k];
799  if (j != k) // If columns are different
800  for (i = 0; i < 4; i++)
801  {
802  hold = -mat.vv[i][k];
803  mat.vv[i][k] = mat.vv[i][j];
804  mat.vv[i][j] = hold;
805  }
806  // Divide column by minus pivot matalue
807  for (i = 0; i < 4; i++)
808  if (i != k) // Don't touch the pivot entry
809  mat.vv[i][k] /= ( -pvt_matal) ; // (Tricky C syntax for division)
810 
811  // Reduce the matrix
812  for (i = 0; i < 4; i++)
813  {
814  hold = mat.vv[i][k];
815  for (j = 0; j < 4; j++)
816  if ( i != k && j != k ) // Don't touch pivot.
817  mat.vv[i][j] += hold * mat.vv[k][j];
818  }
819 
820  // Divide row by pivot
821  for (j = 0; j < 4; j++)
822  if (j != k) // Don't touch the pivot!
823  mat.vv[k][j] /= pvt_matal;
824 
825  // Replace pivot by reciprocal (at last we can touch it).
826  mat.vv[k][k] = 1.0f/pvt_matal;
827  }
828 
829  // That was most of the work, one final pass of row/column interchange
830  // to finish
831  for (k = 4-1; k > 0; k--) // Don't need to work with 1 by 1
832  // corner
833  {
834  i = pvt_j[k-1]; // Rows to swap correspond to pivot COLUM4
835  if (i != k-1) // If rows are different
836  for (j = 0; j < 4; j++)
837  {
838  hold = mat.vv[k-1][j];
839  mat.vv[k-1][j] = -mat.vv[i][j];
840  mat.vv[i][j] = hold;
841  }
842 
843  j = pvt_i[k-1]; // Columns to swap correspond to pimatot ROW
844  if (j != k-1) // If columns are different
845  for (i = 0; i < 4; i++)
846  {
847  hold = mat.vv[i][k-1];
848  mat.vv[i][k-1] = -mat.vv[i][j];
849  mat.vv[i][j] = hold;
850  }
851  }
852  delete[] pvt_i;
853  delete[] pvt_j;
854  return mat;
855  }
