170  template< typename T >
171  Vec3< T > getToroidalVector( const Vec3<T>& p1,
172  const Vec3<T>& p2,
173  const T& sizex=static_cast<T>(1.f),
174  const T& sizey=static_cast<T>(1.f),
175  const T& sizez=static_cast<T>(1.f))
176  {
177  float dx = p2.x - p1.x;
178  if (fabs(dx) > sizex / 2) {
179  if (p2.x < sizex / 2) {
180  dx = p2.x - (p1.x - sizex);
181  } else {
182  dx = p2.x - (p1.x + sizex);
183  }
184  }
185  float dy = p2.y - p1.y;
186  if (fabs(dy) > sizey / 2) {
187  if (p2.y < sizey / 2) {
188  dy = p2.y - (p1.y - sizey);
189  } else {
190  dy = p2.y - (p1.y + sizey);
191  }
192  }
193  float dz = p2.z - p1.z;
194  if (fabs(dz) > sizez / 2) {
195  if (p2.y < sizez / 2) {
196  dz = p2.z - (p1.z - sizez);
197  } else {
198  dz = p2.z - (p1.z + sizez);
199  }
200  }
201  return Vec3f(dx, dy, dz);
202  }
